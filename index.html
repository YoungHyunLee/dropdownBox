1. 1. 일반적인 기본 형태.

              var $mySelectBox = $('#isItSelect');
          // select에 change 이벤트를 적용시켜주시면, 일반적인 select처럼 동작합니다.
          $mySelectBox.on('change', function(){
            alert('날 클릭했냐옹!!!');
          });

          // FIRE!!
          $mySelectBox.ctmDropdownBox({width : '200', maxHeight : '200'});

  2. 2. id값 & selected값이 없는 형태

              $('.temp_select_class').ctmDropdownBox({width : '200', maxHeight : '200', theme : 'temp_theme1'});

  3. 3. 프리하게 사용하자! 프리한 형태.

              // 너비등을 주지 않으면, 내부 select박스의 크기만큼, 알아서 사이즈가 정해짐.
          $('.temp_select_class2').ctmDropdownBox();
